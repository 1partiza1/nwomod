# Islamist Rebels # reactionary
###################################
islamist_rebels = {
	icon = 1
	
	area = nation_religion 					# Stays within the religion
	break_alliance_on_win = yes
	
	# <this is the governement in the country> = <this is what the rebels want>
	government = {
		proletarian_dictatorship = islamist_dictatorship
		presidential_dictatorship = islamist_dictatorship
		populist_dictatorship = islamist_dictatorship
		islamist_dictatorship = democracy
		absolute_monarchy = islamist_dictatorship
		nationalist_dictatorship = islamist_dictatorship
		hms_government = islamist_dictatorship
		democracy = islamist_dictatorship
	}

	defection = religion				
	independence = none				
	defect_delay = 18
	
	ideology = islamist
	
	allow_all_cultures = yes
	allow_all_religions = no
	allow_all_ideologies = yes
	
	resilient = yes					# Disbands on retreat
	reinforcing = yes				# Does not receive reinforcements
	general = yes					# Does use generals
	smart = yes						# Does not assault forts, tactically withdraw from battle, etc.
	unit_transfer = yes				# Units do not remain after enforcing demands.
	
	occupation_mult = 1.0			# The amount of strength this rebel gains from taking provinces

	# triggers a general rising, modifier will be multiplied with the countrys army strength and compared to the factions strenght
	will_rise = 
	{
		factor = 0.5
		
		modifier = {
			war = yes
			factor = 0.8
		}
		
		modifier = {
			war_exhaustion = 4
			factor = 0.95
		}
		
		modifier = {
			war_exhaustion = 6
			factor = 0.95
		}
		
		modifier = {
			war_exhaustion = 8
			factor = 0.95
		}
		
		modifier = {
			factor = 0.95
			any_neighbor_country = {
				government = islamist_dictatorship
			}
		}		
	}
	
	# The rebel type with the highest modifier for this pop gets picked
	spawn_chance = {
		factor = 1.02
		
		modifier = {
			factor = 0
			country = { civilized = no }
		}
		modifier = {
			factor = 0
			NOT = {
				is_ideology_enabled = islamist
			}
		}
		modifier = {
			factor = 0
			government = islamist_dictatorship
		}
		modifier = {
			factor = 1.015
			NOT = { political_reform_want = 0.05 }
		}
		modifier = {
			factor = 1.15
			social_reform_want = 0.15 
			NOT = { political_reform_want = 0.15 
			}
		}
		modifier = {
			factor = 0.01
			NOT = { pop_majority_ideology = islamist }
		}
		modifier = {
			factor = 1.3
			pop_majority_ideology = islamist
		}
		modifier = {
			factor = 1.3
			OR = { religion = sunni religion = shiite }
		}		
		modifier = {
			factor = 1.015
			country = { plurality = 10 }
		}
		modifier = {
			factor = 1.03
			country = {
				any_neighbor_country = {
					government = proletarian_dictatorship
				}
			}
		}
		modifier = {
			factor = 1.05
			country = {
				average_militancy = 6
			}
		}
		modifier = {
			factor = 1.05
			country = {
				average_consciousness = 6
			}
		}
		modifier = {
			factor = 0
			location = {
				is_colonial = yes
			}
		}	
		modifier = {
			factor = 0
			NOT = { OR = { religion = sunni religion = shiite } }
		}
	}
 
 	#This is checked for EACH province in the Area of Operations
	movement_evaluation = {
		factor = 1
		modifier = {
			factor = 0.1
			units_in_province = 1
		}
		modifier = {
			factor = 0.001
			controlled_by = REB
		}
		modifier = {
			factor = 1.5
			average_militancy = 4
		}
		modifier = {
			factor = 1.5
			average_militancy = 6
		}
		modifier = {
			factor = 1.5
			average_militancy = 8
		}
		modifier = {
			factor = 2.0
			is_capital = yes
		}
	}
	
	# Country scope
	demands_enforced_trigger = {
		capital_scope = {
			controlled_by = REB
			province_control_days = 60
		}
	}
	
	demands_enforced_effect = {
		political_reform = none_voting
		political_reform = ruling_party_opinion
		political_reform = quality_of_service
		political_reform = censored_publishing
		political_reform = gender_equal
		political_reform = parties_not_allowed
		political_reform = nobody_gets_out
		political_reform = lgbt_outlawed
		political_reform = minorities_oppression
		# more effects than in vanilla..
		add_country_modifier = {
			name = national_confusion
			duration = 365
		}
		prestige_factor = -0.2		
		war_exhaustion = 50
		country_event = 800051
	}
}
